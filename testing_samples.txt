### Good Code (should pass clean)

```python
import json
import logging
import os
import subprocess
import tempfile
from pathlib import Path

import yaml

logger = logging.getLogger(__name__)


def handle_user_request(user_input):
    """Process user input safely."""

    # Validate input
    if not isinstance(user_input, str) or len(user_input) > 1000:
        raise ValueError("Invalid input")

    # Safe deserialization
    with open("config.yml", "r") as f:
        config = yaml.safe_load(f)

    with open("data.json", "r") as f:
        data = json.load(f)

    # Safe subprocess -- no shell, list args
    sanitized = user_input.strip()
    completed = subprocess.run(
        ["grep", "--", sanitized, "/var/log/app.log"],
        capture_output=True,
        text=True,
        timeout=10,
    )

    # Parameterized SQL
    cursor.execute(
        "SELECT * FROM users WHERE name = %s",
        (sanitized,),
    )

    # Safe temp file
    fd, tmppath = tempfile.mkstemp(prefix="myapp_", suffix=".tmp")
    try:
        os.write(fd, completed.stdout.encode())
    finally:
        os.close(fd)

    # Credentials from environment
    api_key = os.environ.get("API_KEY")
    if not api_key:
        raise RuntimeError("API_KEY not set")

    logger.info("Request processed successfully")
    return data

### Good Code (should pass clean)

```javascript
import DOMPurify from "dompurify";

const ALLOWED_ORIGINS = ["https://app.example.com"];

function processUserData(userInput) {
  if (typeof userInput !== "string" || userInput.length > 5000) {
    throw new Error("Invalid input");
  }

  const sanitized = DOMPurify.sanitize(userInput);
  document.getElementById("output").textContent = sanitized;

  const apiKey = process.env.API_KEY;
  if (!apiKey) throw new Error("API_KEY not configured");

  return fetch("/api/proxy", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ query: sanitized }),
  }).then((res) => {
    if (!res.ok) throw new Error("Request failed");
    return res.json();
  });
}

window.addEventListener("message", (event) => {
  if (!ALLOWED_ORIGINS.includes(event.origin)) return;
  processUserData(event.data);
});
```

### Good Code (should pass clean)

```bash
#!/bin/bash
set -euo pipefail

readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly APP_USER="appuser"

main() {
    local config_dir="${SCRIPT_DIR}/config"
    local log_file
    log_file="$(mktemp /tmp/deploy.XXXXXX.log)"

    if [[ ! -d "${config_dir}" ]]; then
        echo "Error: config directory not found" >&2
        exit 1
    fi

    chmod 640 "${config_dir}/settings.yml"
    chown "${APP_USER}:${APP_USER}" "${config_dir}/settings.yml"

    echo "Deployment complete. Log: ${log_file}"
}

main "$@"
